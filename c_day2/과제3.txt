#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

int check(char s[]);

int main() {
    char arr[20];

    printf("음 입력\n");
    scanf("%s", arr);
    printf("\n");

    int result = check(arr);

    switch (result) {
    case 1:
        printf("결과 : ascending\n");
        break;
    case 2:
        printf("결과 : descending\n");
        break;
    case 3:
        printf("결과 : mixed\n");
        break;
    default:
        printf("잘못된 입력입니다.\n");
        break;
    }

    return 0;
}

int check(char s[]) {
    char asc[] = "cdefgabC";
    char desc[] = "Cbagfedc";

    int as = 1;
    int des = 1;

    for (int i = 0; i < 8; i++) {
        if (s[i] != asc[i]) {
            as = 0;
        }
        if (s[i] != desc[i]) {
            des = 0;
        }
    }

    if (as == 1) {
        return 1;
    }
    else if (des == 1) {
        return 2;
    }
    else {
        return 3;
    }
}
