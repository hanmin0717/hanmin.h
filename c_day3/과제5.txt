#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>

int main() {
	int n;
	int a = 1; //1열의 숫자들의 대각선에 숫자를 부여하기 위한 변수
	int cal = 0;

	printf("숫자를 입력하세요 : ");
	scanf("%d", &n);
	int b = n * n; // 마지막열의 숫자들의 대각선에 숫자를 부여하기 위한 변수
	int* p = (int*)malloc(sizeof(int) * n * n); // 배열처럼 사용하기 위해 n*n을 사용


	for (int i = 0; i < n; i++) { //1열의 숫자들을 기준으로 대각선에 존재하는 변수에 숫자를 부여하기 위한 작업
		for (int j = 0; j <= i; j++) {
			p[i+ (n-1)*j] = a++;
		}
	}

	for (int i = 0; i < n - 1; i++) { //마지막 열을 기준으로 대각선에 존재하는 변수에 숫자를 부여하기 위한 작업
		for (int j = 0; j <= i; j++) {
			p[n*n-1- i - (n - 1) * j] = b--;
		}
	}

	//마지막 출력
	for (int i = 0; i < n * n; i++) {
		printf("%d\t", p[i]); //가독성을 위해 \t 사용
		if (i == n * (cal + 1) - 1) { // 배열의 번호가 n(cal+1)-1과 같다면 줄바꿈을 해주는 작업
			printf("\n");
			cal++;
		}
	}

	free(p); //사용한 동적할당을 풀어주는 작업
	

	return 0;
}
