#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>

int main() {
	int n;
	int a = 1; //1열의 숫자들의 대각선에 숫자를 부여하기 위한 변수
	int cal = 0;

	printf("숫자를 입력하세요 : ");
	scanf("%d", &n);
	if (n <=0) {0또는 그 이하의 숫자를 입력하면 프로그램 종료 
		printf("잘못입력하셨습니다\n");
	}

	else {
		int b = n * n; // 마지막열의 숫자들의 대각선에 숫자를 부여하기 위한 변수
		int* p = (int*)malloc(sizeof(int) * (n * n)); // 배열처럼 사용하기 위해 n*n을 사용


		int num = 1;

		// k는 행과 열의 합
		for (int k = 0; k <= 2 * n - 2; k++) {
			for (int r = 0; r < n; r++) {
				int c = k - r; // 열 좌표 계산

				//값의 범위를 벗어나지 않는 안전한 좌표일 때만 값 대입
				if (c >= 0 && c < n) {
					p[r * n + c] = num++; // 2차원 좌표를 1차원 인덱스로 변환해 대입
				}
			}
		}

		//마지막 출력
		for (int i = 0; i < n * n; i++) {
			printf("%d\t", p[i]); //가독성을 위해 \t 사용
			if (i == n * (cal + 1) - 1) { // 배열의 번호가 n(cal+1)-1과 같다면 줄바꿈을 해주는 작업
				printf("\n");
				cal++;
			}
		}

		free(p); //사용한 동적할당을 풀어주는 작업
	}

	return 0;
}
