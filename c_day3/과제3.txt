#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>

int main() {
	int n1, n2;

	printf("행과 열의 크기를 입력하세요 : ");
	//하나라도 0 이하의 숫자가 들어오면 안전하게 종료
	if (scanf("%d %d", &n1, &n2) != 2 || n1 <= 0 || n2 <= 0) {
		printf("잘못입력하셨습니다\n");
		return 0;
	}

	int* p = (int*)malloc(sizeof(int) * n1 * n2);

	int num = 1;
	int a1 = 0, b1 = -1; // 시작 위치
	int d = 1;     // 이동 방향 플래그 

	int a2 = n1;  // 남은 행의 개수
	int b2 = n2;  // 남은 열의 개수

	while (a2 > 0 && b2 > 0) {
		// 가로 이동
		for (int i = 0; i < b2; i++) {
			b1 += d;
			p[a1 * n2 + b1] = num++;
		}
		a2--; // 가로로 한 줄 채웠으므로 남은 행의 개수 1 감소
		if (a2 <= 0) break;

		// 세로 이동
		for (int i = 0; i < a2; i++) {
			a1 += d;
			p[a1 * n2 + b1] = num++;
		}
		b2--; // 세로로 한 줄 채웠으므로 남은 열의 개수 1 감소

		// 한 바퀴 돌았으므로 방향 뒤집기 
		d *= -1;
	}

	// 마지막 출력 
	int cal = 0;
	for (int i = 0; i < n1 * n2; i++) {
		printf("%d\t", p[i]);
		if (i == n2 * (cal + 1) - 1) {
			printf("\n");
			cal++;
		}
	}

	free(p); // 동적 할당 해제
	return 0;
}
