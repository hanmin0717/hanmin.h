#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

int main() {
    int num[12]; //모양이 3*4 모양이므로 정수형 배열을 12까지 선언
    int q = 1; //반복문을 제한하거나 위에 정수형 배열에 대입을 하기 위해 사용하는 변수

    while (q <=12) { //num이 12개의 배열이므로 q가 12보다 작거나 같을 때만 반복문이 작동하도록 제한을 두었다
        if (q == 1) printf("input : "); // 처음에는 next input이 아니라 input으로 해야함으로 q==1일 때 라는 조건을 달고 아래에는 시작하는 것이 아니기 때문에 next input을 출력하게 만들었다
        else printf("next input : ");

        scanf("%d", &num[q-1]); //숫자를 12번 새롭게 받아야함으로 q를 이용하여 숫자를 받아준다

        // 모양의 규칙을 보면 처음 입력은 하나의 입력이 모든 값에 대입되지만 두 번째 입력부터는 인덱스에 q를 나눈 나머지의 인덱스와 i인덱스의 배열의 값이 같아지는 규칙을 수식으로 표현한 코드
        for (int i = 0; i < 12; i++) {
            num[i] = num[i%q];
        }
        //마지막 출력은 포인터로 했다 i가 1씩 증가하고 int는 크기가 4이지만 포인터로 *(num + i)로 하면 다음 인덱스로 넘어간다.
        for (int i = 0; i < 12; i++) {
            printf("%d", *(num+i));
            if ((i + 1) % 4 == 0) printf("\n");//4번 출력되고 줄을 바꾸므로 나머지 연산으로 4번 출력될 때마다 줄을 바꾸는 코드
        }
        q++;
    }
        return 0;
    
}
